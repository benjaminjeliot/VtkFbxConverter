CMAKE_MINIMUM_REQUIRED (VERSION 2.8.3)
PROJECT (VtkFbxConverter)

LIST(APPEND CMAKE_MODULE_PATH "${CMAKE_SOURCE_DIR}")

FIND_PACKAGE(FBX REQUIRED)

FIND_PACKAGE(Boost COMPONENTS filesystem system regex REQUIRED)
FIND_PACKAGE(VTK REQUIRED)
FIND_PACKAGE(GTest)

INCLUDE( ${VTK_USE_FILE} )

INCLUDE_DIRECTORIES (
	.
	${FBX_INCLUDE_DIRS}
)

ADD_EXECUTABLE(VtkFbxConverter main.cxx
	Common.h
	Common.cxx
	VtkFbxHelper.h
	VtkFbxHelper.cxx
	VtkFbxConverter.h
	VtkFbxConverter.cxx
)

SET (LINK_LIBRARIES
	${FBX_LIBRARIES}
	${Boost_LIBRARIES}
	vtkRendering
	vtkIO
)

TARGET_LINK_LIBRARIES (VtkFbxConverter ${LINK_LIBRARIES})

IF (GTEST_FOUND)
	ADD_SUBDIRECTORY (tests)
ENDIF () # GTEST_FOUND
